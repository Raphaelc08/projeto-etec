<body>

  <form id="formEmail" novalidate>
    <label for="email">Seu e-mail:</label>
    <input type="email" id="email" name="email" placeholder="exemplo@email.com" required>
    <div id="mensagemErro" class="erro">Digite um e-mail válido.</div>
    <button type="submit">Enviar</button>
  </form>

  <script>
    const form = document.getElementById('formEmail');
    const email = document.getElementById('email');
    const mensagemErro = document.getElementById('mensagemErro');

    form.addEventListener('submit', function(e) {
      e.preventDefault(); // impede envio

      if (!email.checkValidity()) {
        mensagemErro.classList.add('mostrar');
      } else {
        mensagemErro.classList.remove('mostrar');
        alert('E-mail enviado com sucesso!');
        // Aqui você pode mandar o e-mail ou fazer outra ação
      }
    });

    // Esconde o erro quando o usuário começa a digitar
    email.addEventListener('input', function () {
      mensagemErro.classList.remove('mostrar');
    });
  </script>
